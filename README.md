## Быстрый старт

```python
from pydajet_metadata import OneSMetadata

# Подключаемся к базе 1С
db = OneSMetadata(
    "Host=localhost;Port=5433;Database=MyBase;Username=postgres;Password=secret;"
)

# Читаем все справочники
for name in db.Справочники.все():
    print(f"Справочник: {name}")

# Читаем конкретный справочник
алгоритмы = db.Справочники['ирАлгоритмы'].all()
for row in алгоритмы:
    print(f"{row['Наименование']}: {row['ТекстАлгоритма'][:50]}...")

# Добавляем запись
new_id = db.Справочники['ирАлгоритмы'].Добавить({
    'Наименование': 'Новый алгоритм',
    'ТекстАлгоритма': 'Сообщить("Привет");',
    'ДатаИзменения': datetime.now(),
})

# Изменяем запись
db.Справочники['ирАлгоритмы'].Изменить(new_id, {
    'Наименование': 'Обновлённый алгоритм',
})

# Удаляем запись
db.Справочники['ирАлгоритмы'].Удалить(new_id)

db.close()
```

## Подключение к базе

### PostgreSQL

```python
db = OneSMetadata(
    "Host=localhost;Port=5433;Database=MyBase;Username=postgres;Password=secret;"
)
```

### MS SQL Server

```python
db = OneSMetadata(
    "Server=localhost;Database=MyBase;User Id=sa;Password=secret;",
    data_source="sqlserver"
)
```

### Фильтрация

```python
q = db.Справочники['ирАлгоритмы']

# Равно
rows = q.where(q.Наименование == 'telegram').all()

# Не равно
rows = q.where(q.Наименование != 'telegram').all()

# Содержит
rows = q.where(q.Наименование.like('%telegram%')).all()

# Несколько условий (AND)
rows = q.where(
    q.Наименование == 'telegram',
    q.ДатаИзменения > datetime(2026, 1, 1),
).all()

# OR (через sqlalchemy)
from sqlalchemy import or_
rows = q.where(
    or_(q.Наименование == 'telegram', q.Наименование == 'test')
).all()
```

new_data = pl.DataFrame([

### Массовое обновление

```python
# Читаем
df = bridge.read("Справочники", "ирАлгоритмы")

# Изменяем в DataFrame
df = df.with_columns(
    pl.when(pl.col("Наименование").str.contains("telegram"))
    .then(pl.lit("Обновлённый telegram"))
    .otherwise(pl.col("Наименование"))
    .alias("Наименование")
)

# Записываем обратно (замена всех записей)
bridge.write(df, "Справочники", "ирАлгоритмы", mode="replace")
```

### Аналитика

```python
analytics = bridge.analytics("Справочники", "ирАлгоритмы")

# Цепочка операций
result = (
    analytics
    .filter(pl.col("Наименование").str.contains("telegram"))
    .select("Наименование", "ДатаИзменения")
    .sort("ДатаИзменения", descending=True)
    .collect()
)
print(result)

# Группировка
grouped = analytics.group_by("ДатаИзменения").sort("len", descending=True)
print(grouped.collect())

# JOIN с другой таблицей
left = bridge.analytics("Справочники", "ТемыУведомлений")
right = bridge.analytics("Справочники", "КаналыОтправкиУведомлений")
joined = left.join(right, on="Ссылка").collect()
```

## Динамические модели

### Генерация моделей

```python
from pydajet_metadata import OneSModelGenerator

# Генерируем все модели
gen = OneSModelGenerator(db)

# Получаем модель по имени
Алгоритм = gen.model("Справочники.ирАлгоритмы")
```

### Работа с моделями

```python
# Создание
alg = Алгоритм(
    Наименование="Тестовый",
    ТекстАлгоритма='Сообщить("Привет");',
    ДатаИзменения=datetime.now(),
)
alg.save()
print(f"ID: {alg.Ссылка}")

# Загрузка из БД
alg2 = Алгоритм.from_db(alg.Ссылка)
print(f"Наименование: {alg2.Наименование}")

# Изменение
alg2.Наименование = "Обновлённый"
alg2.save()

# Удаление
alg2.delete()

# Все записи
for a in Алгоритм.all():
    print(a.Наименование)

# Фильтрация
for a in Алгоритм.where(Алгоритм._query.Наименование == 'telegram'):
    print(a.Наименование)

# Количество
print(f"Всего: {Алгоритм.count()}")

# Сериализация в JSON
json_str = alg.model_dump_json()
print(json_str)
```

## Транзакции


```python
# ─── Примеры использования транзакций ─────────────────

if __name__ == "__main__":
    db = OneSMetadata(
        "Host=localhost;Port=5433;Database=MessageCenter;Username=postgres;Password=qwaseD12;"
    )
    
    # ─── Простая транзакция ───────────────────────────
    with db.transaction():
        # Обе операции либо выполнятся, либо откатятся
        db.Справочники['ирАлгоритмы'].Добавить({
            'Наименование': 'Алгоритм 1',
            'ТекстАлгоритма': 'Сообщить("1");',
            'ДатаИзменения': datetime.now(),
        })
        db.Документы['Уведомления'].Добавить({
            'Дата': datetime.now(),
            'Номер': '0000001',
            'Сообщение': 'Уведомление 1',
        })
    # Автоматический commit
    
    # ─── Транзакция с откатом ─────────────────────────
    try:
        with db.transaction():
            db.Справочники['ирАлгоритмы'].Добавить({
                'Наименование': 'Алгоритм 2',
                'ТекстАлгоритма': 'Сообщить("2");',
                'ДатаИзменения': datetime.now(),
            })
            raise Exception("Что-то пошло не так")
    except Exception:
        print("Транзакция откатилась, Алгоритм 2 не сохранился")
    
    # ─── Вложенная транзакция ─────────────────────────
    with db.transaction():
        # Основная операция
        db.Справочники['ирАлгоритмы'].Добавить({
            'Наименование': 'Алгоритм 3',
            'ТекстАлгоритма': 'Сообщить("3");',
            'ДатаИзменения': datetime.now(),
        })
        
        try:
            with db.nested_transaction():
                # Вложенная операция
                db.Документы['Уведомления'].Добавить({
                    'Дата': datetime.now(),
                    'Номер': '0000002',
                    'Сообщение': 'Уведомление 2',
                })
                raise Exception("Ошибка в документе")
        except Exception:
            print("Документ откатился, но справочник сохранился")
    
    # ─── Точки сохранения ─────────────────────────────
    with db.transaction():
        db.Справочники['ирАлгоритмы'].Добавить({
            'Наименование': 'Алгоритм 4',
            'ТекстАлгоритма': 'Сообщить("4");',
            'ДатаИзменения': datetime.now(),
        })
        
        # Создаём точку сохранения
        sp = db.savepoint("после_алгоритма_4")
        
        try:
            db.Документы['Уведомления'].Добавить({
                'Дата': datetime.now(),
                'Номер': '0000003',
                'Сообщение': 'Уведомление 3',
            })
        except Exception:
            # Откатываем только документ
            db.rollback_to(sp)
            print("Откатили документ, справочник сохранился")
    
    db.close()
```

**Добавленные методы:**

| Метод | Описание |
|---|---|
| `transaction()` | Контекстный менеджер транзакции (commit/rollback) |
| `nested_transaction()` | Вложенная транзакция (savepoint) |
| `savepoint(name)` | Создать точку сохранения |
| `rollback_to(name)` | Откатиться к точке сохранения |
| `commit()` | Ручной commit |
| `rollback()` | Ручной rollback |

**Особенности:**
- `transaction()` — атомарность: либо все операции, либо ничего
- `nested_transaction()` — вложенный savepoint, можно откатить часть операций
- `savepoint()` — ручное управление точками сохранения
- При выходе из `with` без ошибок — автоматический commit
- При исключении — автоматический rollback

## API Reference

### OneSMetadata

| Метод/Свойство | Описание |
|---|---|
| `__init__(connection_string, data_source)` | Подключение к БД |
| `типы` | Список всех типов объектов |
| `объекты` | Словарь всех объектов по типам |
| `Справочники['имя']` | Доступ к объекту |
| `Документы['имя']` | Доступ к объекту |
| `close()` | Закрыть соединение |

### TableQuery

| Метод | Описание |
|---|---|
| `all()` | Все записи |
| `first()` | Первая запись |
| `count()` | Количество записей |
| `where(*conditions)` | Условие WHERE |
| `Добавить(data)` | Добавить запись |
| `Изменить(id, data)` | Изменить запись |
| `Удалить(id)` | Удалить запись |
| `ДобавитьСЧастями(data, части)` | Добавить с табличными частями |
| `ТабличныеЧасти` | Список имён табличных частей |
| `Часть(name)` | Доступ к табличной части |

### OneSPolarsBridge

| Метод | Описание |
|---|---|
| `read(type_name, object_name)` | Читать таблицу в Polars DataFrame |
| `write(df, type_name, object_name, mode)` | Записать DataFrame в БД |
| `analytics(type_name, object_name)` | Получить объект Analytics |

### OneSModelGenerator

| Метод | Описание |
|---|---|
| `model(name)` | Получить модель по имени |
| `get_all_models()` | Все сгенерированные модели |
